```python
# agents/groq_agent.py

from groq import Groq

class GroqPoweredAgent:
    def __init__(self):
        self.client = Groq(api_key="gsk_...")  # FREE
        self.model = "mixtral-8x7b-32768"  # FREE
    
    def scout_agent(self, location: str, category: str):
        """Scout Agent powered by Groq (FREE)"""
        
        message = self.client.chat.completions.create(
            model=self.model,
            messages=[
                {
                    "role": "system",
                    "content": "You are Scout Agent..."
                },
                {
                    "role": "user",
                    "content": f"Find {category} in {location}"
                }
            ],
            max_tokens=1024,
            temperature=0.7
        )
        
        return message.choices[0].message.content
    
    def closer_agent(self, lead: dict):
        """Closer Agent powered by Groq (FREE)"""
        
        message = self.client.chat.completions.create(
            model=self.model,

                    "content": "You are Closer Agent..."
                },
                {
                    "role": "user",
                    "content": f"Craft message for {lead['name']}"
                }
            ],
            max_tokens=512,
            temperature=0.8
        )
        
        return message.choices[0].message.content

        """Architect Agent powered by Groq (FREE)"""

            model=self.model,

                {
```
